üí° Usage Examples

JavaScript

const ws = new WebSocket('ws://localhost:8000/chats/ws?token=' + token); ws.onopen = () => { ws.send(JSON.stringify({ action: 'subscribe', chat_ids: ['chat-uuid-1'] })); }; ws.onmessage = (event) => { const data = JSON.parse(event.data); console.log('Received:', data); };

Python

import asyncio import websockets import json async with websockets.connect(uri) as websocket: await websocket.send(json.dumps({ "action": "subscribe", "chat_ids": ["chat-uuid-1"] })) async for message in websocket: data = json.loads(message) print(f"Received: {data}")
